module calc lines 
Fonction Lines (@fs : text):entier 
begin
    ouvrir("compte.txt", fs, "r")
    n <- 0 
    quit = faux 
    tant ques quit = faux faire 
        lire_ln(fs, ele) 
        si long(ele) > 0 alors 
            n <- n+1 
        sinon 
            quit <- vrai 
        fin si 

    fin  tant que 
    fermer(fs)
    retourner n 
end

module exist 

Fonction Exist (@fs , ele: chaine):booleen
begin   
    ouvrir("compte.txt", @fs,  "r")
    pour i de 1 a Lines(fs) faire

            lire_ln(fs, a)
            si a = ele alors 
                retourner vrai
            fin si 
        fin si 

    fin pour 

    fermer(fs)
    retourner faux 
fin

procedure fillRes(n:enteir, @fv:fiche, @fs:text)
debut 
    pour i de 1 a n faire 
        lire(fv, bank)
        trans <- 0
        pour j de 1 a n faire 
            lire(fv , b)
            si i != j faire 
                si bank.num = b.num alors 
                    si b.num = "D" alors 
                        trans <- trans - b.mont 
                    sinon 
                        trans <- trans + b.mont
                    sin si 
                fin si 
            fin si 
        fin pour 
        si bank.nat = "D" alors 
                trans <- trans - bank.mont 
        sinon 
            trans <- trans + bank.mont
