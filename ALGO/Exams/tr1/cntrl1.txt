module calc lines 
Fonction Lines (@fs : text):entier 
begin
    ouvrir("compte.txt", fs, "r")
    n <- 0 
    quit = faux 
    tant ques quit = faux faire 
        lire_ln(fs, ele) 
        si long(ele) > 0 alors 
            n <- n+1 
        sinon 
            quit <- vrai 
        fin si 

    fin  tant que 
    fermer(fs)
    retourner n 
end

    n : entier
    quit : booleen 
    ele : chaine 
}

fin

procedure Sort(t:t) 
debut 
    pour i de 0 a long(t)-2 faire
        si t[i].mont > t[i+1].mont alors 
            aux <- t[i]
            t[i]<- t[i+1]            
            t[i+1] <- aux 
            Sort(t)
fin

procedure fillRes(n:enteir, @fv:fiche, @fs:text)
debut 
    ouvrir("mvt.dat", fv, "rb")
    ouvrir("result.txt", fs, "w")
    
    pour i de 1 a n faire 
        lire(fv, bank)
        trans <- 0
        pour j de 1 a n faire 
            lire(fv , b)
            si i != j faire 
                si bank.num = b.num alors 
                    si b.num = "D" alors 
                        trans <- trans - b.mont 
                    sinon 
                        trans <- trans + b.mont
                    sin si 
                fin si 
            fin si 
        fin pour 
        si bank.nat = "D" alors 
                trans <- trans - bank.mont 
        sinon 
            trans <- trans + bank.mont 
        fin si 

        sol.num <- bank.num 
        sol.mont <- trans
        t[i] <- sol



        
    fin pour 
    Sort(t)
    pour i de 0 a n-1 faire 
        ecrire_nl(fs, t[i].num + " " + convch(t[i].mont))
    fin pour 

    fermer(fs)
    fermer(fv)


fin
